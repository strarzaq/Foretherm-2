APPS SCRIPT SETUP INSTRUCTIONS

1. Create a Google Sheet with this header in row 1:

timestamp site date measured_2m height_offset estimated_cable era5_raw forecast_mean forecast_cold forecast_warm error_2m error_cable notes

2. Extensions → Apps Script → paste this code → Deploy → New deployment → Web app → Execute as Me → Anyone → Deploy → copy URL:

function doGet() { var s = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var rows = s.getDataRange().getValues(); var h = rows[0]; var data = rows.slice(1).map(function(r) { var o = {}; h.forEach(function(k,i){ o[k]=r[i]; }); return o; }); return ContentService.createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { var s = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var data = JSON.parse(e.postData.contents); var h = s.getRange(1,1,1,s.getLastColumn()).getValues()[0]; var last = s.getLastRow(); var sc = h.indexOf('site')+1, dc = h.indexOf('date')+1; var target = -1; if (last > 1) { var sv = s.getRange(2,sc,last-1,1).getValues(); var dv = s.getRange(2,dc,last-1,1).getValues(); for (var i=0;i<sv.length;i++) { if (sv[i][0]===data.site && dv[i][0]===data.date){target=i+2;break;} } } var row = h.map(function(k){ return data[k]!=null?data[k]:''; }); if (target>0) s.getRange(target,1,1,row.length).setValues([row]); else s.appendRow(row); return ContentService.createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); }
